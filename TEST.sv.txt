`ifndef TEST
`define TEST
`include "interface.sv"
`include "env.sv"

class test;
  env e1;
  virtual intf inf;
  int testcase_id;

  function new(virtual intf inf);
    this.inf = inf;
    e1 = new(inf);
  endfunction

  task run();
    // Get testcase_id from command line argument
    if (!$value$plusargs("TESTCASE=%d", testcase_id))
      testcase_id = 2; // default is write operation



    // Assign test case to environment
    e1.testcase_select = testcase_id;

    // Start environment
    e1.run();
  endtask
endclass

`endif
